def initialize(alarm_name, instance = nil)
  check_init_arg 'alarm_name', 'CloudWatch::Alarm', alarm_name
  @alarm_name = alarm_name
  @aws = instance.nil? ? Aws::CloudWatch::Client.new : instance
  get_alarm alarm_name
end
