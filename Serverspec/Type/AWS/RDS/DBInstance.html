def initialize(dbi_name, instance = nil)
  check_init_arg 'dbi_name', 'RDS::DBInstance', dbi_name
  @dbi_name = dbi_name
  @aws = instance.nil? ? Aws::RDS::Client.new : instance
  get_dbi dbi_name
end
