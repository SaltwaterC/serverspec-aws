def initialize(tbl_name, instance = nil)
  check_init_arg 'tbl_name', 'DynamoDB::Table', tbl_name
  @tbl_name = tbl_name
  @aws = instance.nil? ? Aws::DynamoDB::Client.new : instance
  get_table tbl_name

  @type_map = {
    'S' => :string,
    'N' => :number,
    'B' => :binary
  }
end
