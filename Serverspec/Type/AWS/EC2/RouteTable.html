def initialize(rtb_id, instance = nil)
  check_init_arg 'rtb_id', 'EC2::RouteTable', rtb_id
  @rtb_id = rtb_id
  @aws = instance.nil? ? Aws::EC2::Client.new : instance
  get_rtb rtb_id
end
