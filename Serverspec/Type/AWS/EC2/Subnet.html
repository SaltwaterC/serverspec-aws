def initialize(subnet_id, instance = nil)
  check_init_arg 'subnet_id', 'EC2::Subnet', subnet_id
  @subnet_id = subnet_id
  @aws = instance.nil? ? Aws::EC2::Client.new : instance
  get_subnet subnet_id
end
