def initialize(igw_id, instance = nil)
  check_init_arg 'igw_id', 'EC2::InternetGateway', igw_id
  @igw_id = igw_id
  @aws = instance.nil? ? Aws::EC2::Client.new : instance
  get_igw igw_id
end
