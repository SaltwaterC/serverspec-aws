def initialize(sg_id, instance = nil)
  check_init_arg 'sg_id', 'EC2::SecurityGroup', sg_id
  @sg_id = sg_id
  @aws = instance.nil? ? Aws::EC2::Client.new : instance
  get_security_group sg_id
end
