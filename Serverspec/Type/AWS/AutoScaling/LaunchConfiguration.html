def initialize(config_name, instance = nil)
  check_init_arg(
    'config_name',
    'AutoScaling::LaunchConfiguration',
    config_name
  )
  @config_name = config_name
  @aws = instance.nil? ? Aws::AutoScaling::Client.new : instance
  get_config config_name
end
