def initialize(stack_name, instance = nil)
  check_init_arg 'stack_name', 'CloudFormation::Stack', stack_name
  @stack_name = stack_name
  @aws = instance.nil? ? Aws::CloudFormation::Client.new : instance
  get_stack stack_name
end
