def initialize(cluster_id, instance = nil)
  check_init_arg 'cluster_id', 'Redshift::Cluster', cluster_id
  @cluster_id = cluster_id
  @aws = instance.nil? ? Aws::Redshift::Client.new : instance
  get_cluster cluster_id
end
